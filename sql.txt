use coach360;
CREATE TABLE `users` (
    `id` int(10) NOT NULL auto_increment PRIMARY KEY,
    `name` varchar(255) COLLATE utf8_unicode_ci NOT NULL,
    `email` varchar(55) NOT NULL,    
    `password` varchar(40) NOT NULL,
    `phone_number` varchar(15),
    `sex` int(1) NOT NULL,    
    `birth_day` date,
    `field_work` varchar(150) COLLATE utf8_unicode_ci,
    `province` int(2),
    `address` varchar(255) COLLATE utf8_unicode_ci,
    `question_number` int(5) DEFAULT 0,
    `avatar` text COLLATE utf8_unicode_ci,
    `securi_code` varchar(8) DEFAULT "TIPOCHO",
    `is_admin` int(1) DEFAULT 0,
    `status` int(1)
)

CREATE TABLE `provincese` (
    `id` int(10) NOT NULL auto_increment,
    `province_name` nvarchar(255),
    PRIMARY KEY (`id`)
)

CREATE TABLE `erorrs` (
	`id` int(10) PRIMARY KEY NOT NULL auto_increment,
	`erorr` nvarchar(255)
)

CREATE TABLE `questions` (
    `id` int(10) NOT NULL auto_increment PRIMARY KEY,
    `user` int(10),
    `title` nvarchar(255),    
    `content` longtext COLLATE utf8_unicode_ci,
    `status` int(1) DEFAULT 0
)

CREATE TABLE `reply_questions` (
    `id` int(10) NOT NULL auto_increment PRIMARY KEY,
    `question` int(10),  
    `content` longtext COLLATE utf8_unicode_ci
)

CREATE TABLE `blog_post` (
    `id` int(10) NOT NULL auto_increment PRIMARY KEY,
    `category` int(10),
    `author` int(10),
    `title` nvarchar(255), 
    `content` longtext COLLATE utf8_unicode_ci,
    `view` int(10)
)

CREATE TABLE `videos` (
    `id` int(10) NOT NULL auto_increment PRIMARY KEY,
    `blog_post` int(10),
    `title` nvarchar(255), 
    `link` text COLLATE utf8_unicode_ci,
    `content` text COLLATE utf8_unicode_ci
)


CREATE TABLE `images` (
    `id` int(10) NOT NULL auto_increment PRIMARY KEY,
    `blog_post` int(10),
    `title` nvarchar(255), 
    `link` text COLLATE utf8_unicode_ci,
    `content` text COLLATE utf8_unicode_ci
)
